def euclidean_distance(point1, point2):
    return np.linalg.norm(point1 - point2)

def cosine_distance(point1, point2):
    return 1 - np.dot(point1, point2) / (np.linalg.norm(point1) * np.linalg.norm(point2))

def manhattan_distance(point1, point2):
    return np.sum(np.abs(point1 - point2))

def create_count_table(y_train, labels):
    
    df = pd.DataFrame({'True_Label': y_train, 'Cluster_Label': labels})

    count_table = pd.crosstab(df['Cluster_Label'], df['True_Label'])

    return count_table

# Function to assign labels to clusters based on the majority class in each cluster
def assign_labels_to_clusters(count_table):
    num_clusters, num_labels = count_table.shape
    cluster_labels = [None] * num_clusters  # Initialize an array to hold cluster labels
    used_labels = set()  # Initialize a set to track used labels

    # Iterate until all clusters are labeled
    while None in cluster_labels and len(used_labels) < num_labels:
        # Find the maximum value in the count table and its indices
        max_value = -1
        max_i = -1
        max_j = -1
        for i in range(num_clusters):
            for j in range(num_labels):
                if count_table.iloc[i, j] > max_value and cluster_labels[i] is None and j not in used_labels:
                    max_value = count_table.iloc[i, j]
                    max_i = i
                    max_j = j
        
        # If a valid maximum was found, assign the label
        if max_i != -1 and max_j != -1:
            cluster_labels[max_i] = max_j  # Assign label j to cluster i
            used_labels.add(max_j)  # Add label j to the set of used labels

    return cluster_labels

accuracies_train = []
confusion_matrices_train = []
accuracies_test = []
confusion_matrices_test = []
count = 0

for i in range(5):

    # Split the data into 80% training and 20% testing
    x_train_split, x_test_split, y_train_split, y_test_split = train_test_split(x_all_scaled, y_all, test_size=0.2, random_state=42)

    # Define the distance metric
    distance_metric_choice = distance_metric(type_metric.USER_DEFINED, func=euclidean_distance)

    # Number of components
    num_clusters = 10

    # Initialize the centers for KMeans using KMeans++
    initial_centers = kmeans_plusplus_initializer(x_train_split, num_clusters).initialize()

    # Create a KMeans object using pyclustering with the chosen distance metric
    kmeans_instance = kmeans(x_train_split, initial_centers, metric=distance_metric_choice)

    # Perform KMeans clustering
    kmeans_instance.process()

    # Retrieve the cluster labels and centers
    clusters = kmeans_instance.get_clusters()
    centers = kmeans_instance.get_centers()

    # Create a list of labels for each data point
    labels = np.zeros(x_train_split.shape[0], dtype=int)
    for cluster_id, cluster_points in enumerate(clusters):
        for point_index in cluster_points:
            labels[point_index] = cluster_id

    # Call count table function
    count_table = create_count_table(y_train_split, labels)
    print("----------------------------------------------")
    print(f'Table {count+1} (Cluster Label vs True Label):')
    print(count_table)

    # Call assigning function
    cluster_labels = assign_labels_to_clusters(count_table)

    print("\nCluster Labels:", cluster_labels)
    print("----------------------------------------------")
    # Create a list of assigned labels for each data point
    labels_assigned = np.zeros(x_train_split.shape[0], dtype=int)
    for cluster_id, cluster_points in enumerate(clusters):
        for point_index in cluster_points:
            labels_assigned[point_index] = cluster_labels[cluster_id]
            
    # Calculate the normalized confusion matrix
    conf_matrix_train = confusion_matrix(y_train_split, labels_assigned, normalize='true')
    confusion_matrices_train.append(conf_matrix_train)
  
    # Calculate the accuracy
    accuracy_train = accuracy_score(y_train_split, labels_assigned)
    accuracies_train.append(accuracy_train)
    

    #### TEST DATA ####

    # Use 1-NN to classify each data point in the test set
    knn_classifier = KNeighborsClassifier(n_neighbors=1)
    
    # Fit the 1-NN classifier using the cluster centers as training data
    knn_classifier.fit(centers, cluster_labels)

    # Predict the cluster for each test data point
    predicted_clusters = knn_classifier.predict(x_test_split)

    # Calculate the normalized confusion matrix
    conf_matrix_test = confusion_matrix(y_test_split, predicted_clusters, normalize='true')
    confusion_matrices_test.append(conf_matrix_test)

    # Calculate the accuracy
    accuracy_test = accuracy_score(y_test_split, predicted_clusters)
    accuracies_test.append(accuracy_test)

    count += 1
